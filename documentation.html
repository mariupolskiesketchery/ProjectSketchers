 28@app.route('/upload', methods=['POST'])
 29def upload_file():
 30    if 'image' not in request.files:
 31        return "Ошибка: изображение не загружено"
 32
 33    file = request.files['image']
 34    if file.filename == '':
 35        return "Ошибка: файл не выбран"
 36
 37    img = Image.open(file)

 39
 40    # Обнаружение объектов с использованием йоло

 42    detected_objects = []

 44    for det in results.xyxy[0]:

 49
 50    if not detected_objects:
 51        detected_objects = ["Объекты не найдены"]
 52
 53    # Извлечение текста
 54    extracted_text = pytesseract.image_to_string(img, lang='rus')
 55    corrected_text = correct_text_with_yandex_speller(extracted_text)
 56
 57    # Проверка на ошибки
 58    text_message = "Ошибок не найдено" if extracted_text.strip() == corrected_text.strip() else corrected_text
 59
 60    # Подготовка изображений для отображения

 62
 63    return render_template('result.html',
 64                           original_image=original_image_base64,
 65                           detected_objects=", ".join(detected_objects),

 86def correct_text_with_yandex_speller(text):
 87    response = requests.post(YANDEX_SPELLER_URL, data={'text': text}).json()
 88
 89    if not response:
 90        return text  # Если ошибок нет, возвращаем старый текст
 91
 92    corrected_text = text
 93    for item in response:
 94        if 's' in item and item['s']:
 95            corrected_text = corrected_text.replace(item['word'], item['s'][0])
 96
 97    return corrected_text

107def search_images_by_metadata(folder_path, search_word):
108    matching_images = []
109    for filename in os.listdir(folder_path):
110        if filename.lower().endswith(('png', 'jpg', 'jpeg')):
111            image_path = os.path.join(folder_path, filename)
112            image = Image.open(image_path)
113            exif_data = image.getexif()
114
115            if 0x9286 in exif_data:
116                metadata = exif_data[0x9286].decode('utf-8').lower()
117                if search_word.lower() in metadata:
118                    matching_images.append(filename)
119
120    return matching_images
